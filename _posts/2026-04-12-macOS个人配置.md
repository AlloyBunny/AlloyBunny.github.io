## 体验优化类

### LinearMouse

优化鼠标行为：反向滚轮、线性滚轮、滚轮滚动行数设置、线性指针移动

### DockDoor

像windows一样的程序坞窗口预览、Alt+Tab切屏。

### Rectangle

像windows一样的分屏窗口吸附

### 超级右键

像windows一样的右键，能直接新建文件（但似乎无法实现“用cursor打开当前目录”之类的功能）

### Bartender 5

隐藏部分菜单栏，防止显示不全

## 实用工具类

### utools

快捷键设置成F3，装“文件搜索”、“剪切板”、“OCR Pro”

F4绑定成OCR识别文字

### Pixpin

F1绑定截图，F2绑定贴图

### 豆包输入法

截止2026.04.12，该输入法还是内测版本，google上能找到。它的语音输入功能完全免费，且延迟非常低，功能很强大，非常推荐。

## 深度定制类

### Hammerspoon

自己写脚本，实现在按住空格的情况下，a是←，s是→，d是↑，f是↓，j是comand+←，k是command+→，l是comand+shift+←，;是command+shift+→，退格键是delete。脚本放在~/.hammerspoon/init.lua，参考脚本如下：

```lua
-- =========================================================
-- Spacebar Modifier (稳定版)
-- 目标：
-- 1. 点按 Space => 正常输入空格
-- 2. 按住 Space + A/S/D/F/J/K/L/; => 导航
-- 3. 一旦 Space 期间触发过映射或组合，松开 Space 时不再输出空格
-- 4. 支持长按 A/S/D/F 连续移动
-- 5. 防止先松 Space 再松映射键时出现“卡键”
-- =========================================================

local eventtypes = hs.eventtap.event.types
local eventprops = hs.eventtap.event.properties
local keymap = hs.keycodes.map

-- 映射表
-- 你原先的定义我保留了：
-- A 左, S 右, D 上, F 下
-- J cmd+left, K cmd+right
-- L cmd+shift+left, ; cmd+shift+right
local mappings = {
    [keymap["a"]]      = { key = "left",  mods = {} },
    [keymap["s"]]      = { key = "right", mods = {} },
    [keymap["d"]]      = { key = "up",    mods = {} },
    [keymap["f"]]      = { key = "down",  mods = {} },
    [keymap["j"]]      = { key = "left",  mods = { "cmd" } },
    [keymap["k"]]      = { key = "right", mods = { "cmd" } },
    [keymap["l"]]      = { key = "left",  mods = { "cmd", "shift" } },
    [keymap[";"]]      = { key = "right", mods = { "cmd", "shift" } },
    [keymap["delete"]] = { key = "forwarddelete", mods = {} },
}

-- 状态变量
local spaceDown = false          -- 物理空格当前是否按下
local spaceUsed = false          -- 本次空格按住期间是否被当作“修饰层”使用过
local activeVirtualKeys = {}     -- 记录哪些物理键当前对应着一个“虚拟按下”的目标键
local suppressSelfEvents = false -- 防止我们自己补发的事件又被自己拦截

local spaceTap = nil

-- 更稳的“补发一个空格”
-- 不用 hs.eventtap.keyStroke，而是手动发 keyDown + keyUp
local function emitSpace()
    suppressSelfEvents = true
    spaceTap:stop()

    hs.eventtap.event.newKeyEvent({}, "space", true):post()
    hs.eventtap.event.newKeyEvent({}, "space", false):post()

    spaceTap:start()
    suppressSelfEvents = false
end

-- 发一个新的目标按键事件，必要时带 autorepeat 属性
local function makeMappedEvent(map, isDown, isAutoRepeat)
    local ev = hs.eventtap.event.newKeyEvent(map.mods, map.key, isDown)
    if isDown and isAutoRepeat then
        ev:setProperty(eventprops.keyboardEventAutorepeat, 1)
    end
    return ev
end

spaceTap = hs.eventtap.new({ eventtypes.keyDown, eventtypes.keyUp }, function(e)
    if suppressSelfEvents then
        return false
    end

    local keyCode = e:getKeyCode()
    local isDown = (e:getType() == eventtypes.keyDown)
    local isAutoRepeat = (e:getProperty(eventprops.keyboardEventAutorepeat) or 0) ~= 0

    -- =========================================================
    -- 1. 处理空格本体
    -- =========================================================
    if keyCode == keymap["space"] then
        local flags = e:getFlags()

        -- 如果此时还按着系统修饰键，则不接管，让原生快捷键正常工作
        -- 例如 Cmd+Space / Ctrl+Space / Option+Space 等
        if flags.cmd or flags.ctrl or flags.alt or flags.shift then
            return false
        end

        if isDown then
            -- 丢弃空格自身的自动连发
            if isAutoRepeat then
                return true
            end

            spaceDown = true
            spaceUsed = false
            return true -- 拦截真实 Space down，先不输出
        else
            -- Space up
            spaceDown = false

            -- 只有“这次空格完全没被当成层键使用”时，才补发真正的空格
            if not spaceUsed then
                emitSpace()
            end

            return true -- 拦截真实 Space up
        end
    end

    -- =========================================================
    -- 2. 当空格层激活时，处理映射键的 KeyDown / Repeat
    -- =========================================================
    if spaceDown and isDown then
        local map = mappings[keyCode]

        if map then
            spaceUsed = true

            -- 第一次按下时登记，后续 autorepeat 不重复登记
            if not activeVirtualKeys[keyCode] then
                activeVirtualKeys[keyCode] = map
            end

            local newEvent = makeMappedEvent(map, true, isAutoRepeat)
            return true, { newEvent }
        else
            -- 按住空格时按了非映射键：
            -- 视为空格已被“用作修饰层”，因此松开空格时不要再输出空格
            -- 但这个键本身正常放行
            spaceUsed = true
            return false
        end
    end

    -- =========================================================
    -- 3. 处理被我们接管的物理键的 KeyUp
    --    即使这时空格已经先松开，也必须把虚拟 KeyUp 发出去
    -- =========================================================
    if (not isDown) and activeVirtualKeys[keyCode] then
        local map = activeVirtualKeys[keyCode]
        activeVirtualKeys[keyCode] = nil

        local newEvent = makeMappedEvent(map, false, false)
        return true, { newEvent }
    end

    -- =========================================================
    -- 4. 其他按键全部放行
    -- =========================================================
    return false
end)

spaceTap:start()

hs.alert.show("Space layer loaded")

```
